# In[33]:


def categorize_user_score(score):
    if pd.isna(score):
        return 'Unknown'
    elif 8 <= score <= 10:
        return 'Высокая'
    elif 3 <= score < 8:
        return 'Средняя'
    elif 0 <= score < 3:
        return 'Низкая'
    else:
        return 'Unknown'  # на случай ошибок

df_actual['user_score_category'] = df_actual['user_score'].apply(categorize_user_score)


# 

# In[34]:


def categorize_critic_score(score):
    if pd.isna(score):
        return 'Unknown'
    elif 80 <= score <= 100:
        return 'Высокая'
    elif 30 <= score < 80:
        return 'Средняя'
    elif 0 <= score < 30:
        return 'Низкая'
    else:
        return 'Unknown'

df_actual['critic_score_category'] = df_actual['critic_score'].apply(categorize_critic_score)
